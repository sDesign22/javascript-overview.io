1.3 Conditional Statements (Code Example)

Code Examples

var x = "Hello";
var y = "Hello";
var z = "New";
document.write("// This Script code shows Conditional Statements as well as the fundamentals of Comparison and Logical Operators." + "
")
if (3 > 2) { document.write("// This shows a condition as: if (3 > 2) witch is true so this code executes as string (text) data here." + "
") }
if (12 >= 18) { } else { document.write("// This example does the same but with numbers and compares using the >= (Greater than or Equal to) Operator as: if (12 >= 18)" +
"Since this is false the code is executed within else braces." + "
") }
if (2 > 3 && x == y) { document.write("// This Statements show: if (2 > 3 && x == y) - And will only show as true if left comparison and right are true." + "
") } else { document.write("This code did not run.") }
if (2 > 3 || x == y) { document.write("// This Statements show: if (2 > 3 || x == y) - Or will always show as true if left comparison or right are true." + "
") } else { document.write("This code did not run.") }
if (!x == y) { document.write("// This Statements Condition Same") } else { document.write("//This code runs because else is provided; if (!x == y) is now made false." + "
") }
if (3 < 2) { } else if (3 > 2) { document.write("else if is run inside this write statement.") } else { document.write("1234567890") }
switch (z) { case "New": case 21: x = 10; break;
case 11: document.write("// Does not Execute."); break;
case x = 10: document.write("// X was changed to 10."); break;
default: document.write("// Does not execute the Default or this would show in the output document."); break; }
document.write("// The final switch statement uses as its parameter (z) to find in our first case its own value and another to find 21(===); next we use this to execute a new value to x(10)." +
" and now a case statement is made to show case again (11) and does not execute(because false). X Now 10:");
